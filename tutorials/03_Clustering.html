```{python}
# FHS data from MIT
url = "https://ocw.mit.edu/courses/15-071-the-analytics-edge-spring-2017/5d689a024551e672313f7fd7eb1bee8d_framingham.csv"
df = pd.read_csv(url)

print(f'df.shape: {df.shape}')
print(f'df.dtypes: \n{df.dtypes}')
print(f'df.head: \n{df.head()}')

num_cols = df.select_dtypes(include='number').columns

cat_cols = [
    "male",
    "education",
    "BPMeds",
    "currentSmoker",
    "prevalentStroke",
    "prevalentHyp",
    "diabetes",
    "TenYearCHD",
]
```

```{python}
#| label: data_select

df_clean = df.dropna()

ann = df_clean[cat_cols].copy()
X = df_clean.drop(columns=cat_cols).copy()
print(f"X.shape: {X.shape}, ann.shape: {ann.shape}")

# transform
X["cigsPerDay"] = np.sqrt(X["cigsPerDay"])
X["glucose"] = np.log1p(X["glucose"])
X["totChol"] = np.log1p(X["totChol"])
```

```{python}
#| label: data_transform
from sklearn.preprocessing import StandardScaler

scaler = StandardScaler()
Z = scaler.fit_transform(X) # Z: numpy array !
Z = pd.DataFrame(Z, columns=X.columns, index=X.index)
num_norm = ( 
  Z
  .melt(var_name="variable", value_name="value")
)

plt.figure(figsize=FIGSIZE)
sns.violinplot(data=num_norm, x="variable", y="value")
plt.xticks(rotation=90)
plt.ylabel("Z-score")
plt.title("Scaled Violins: Standardized Variables")
plt.tight_layout()
plt.show()
```

```{python}
#| label: hierachical_clustering_dendogram

L = linkage(Z, method="complete", metric="euclidean")

# cut tree and define number of clusters
THRESH=10
cluster_hclust = fcluster(L, t=THRESH, criterion="distance")
n_clusters = len(np.unique(cluster_hclust))

plt.figure(figsize=FIGSIZE)
dendrogram(L, no_labels=True, color_threshold=THRESH)
plt.title(f"Hierarchical clustering. {n_clusters} clusters @ threshold = {THRESH}")
plt.xlabel("Samples")
plt.ylabel("Distance")
plt.tight_layout()
plt.show()
```


```{python}
#| label: hierarchical_clustering_data

# linkages for rows and columns
L_row = linkage(Z, method="complete", metric="euclidean")
L_col = linkage(Z.T, method="complete", metric = "correlation")

THRESH=10
cluster_hclust = fcluster(L_row, t=THRESH, criterion="distance")
n_clusters = len(np.unique(cluster_hclust))
print(f"thresh={THRESH} -> {n_clusters} clusters")
ann["h_cluster"] = pd.Categorical(cluster_hclust)

cluster_palette = dict(zip(
    ann["h_cluster"].unique(),
    sns.color_palette("tab10", n_clusters)
))

diagnosis_palette = dict(zip([1, 0], ["black", "white"]))

row_colors = pd.DataFrame({
    "TenYearCHD": ann["TenYearCHD"].map(diagnosis_palette),
    "Smoker": ann["currentSmoker"].map(diagnosis_palette),
    "h_cluster": ann["h_cluster"].astype(int).map(cluster_palette),

})

g = sns.clustermap(
    Z,
    row_linkage=L_row,
    col_linkage=L_col,
    cmap="vlag",
    center=0,
    cbar_pos=(0.92, 0.25, 0.02, 0.5),
    cbar_kws={"label": "Z"},
    row_colors=row_colors,
    xticklabels=True,
    yticklabels=False,
    figsize=(7, 7)
)

# Move colorbar next to heatmap
heat_pos = g.ax_heatmap.get_position()
g.cax.set_position([
    heat_pos.x1 + 0.015,   # small gap to heatmap
    heat_pos.y0,
    0.02,                 # width
    heat_pos.height
])

plt.show()
```


```{python}
#| label: screen_k

k_values = range(1, 41)
res_kmeans = [
    KMeans(n_clusters=k, n_init=10, random_state=RANDOM_STATE).fit(Z)
    for k in k_values
]

screening = pd.DataFrame({
    "k": list(k_values),
    "within_ss": [model.inertia_ for model in res_kmeans],
})

plt.figure(figsize=FIGSIZE)
sns.lineplot(data=screening, x="k", y="within_ss", marker="o")
plt.title("Within-cluster SS vs k")
plt.xlabel("Number of clusters k")
plt.ylabel("Total within-cluster SS (inertia)")
plt.tight_layout()
plt.show()
```


```{python}
#| label: select_k

from sklearn.cluster import KMeans
km_res = res_kmeans[5]

print("Number of iterations:", km_res.n_iter_)
print("Cluster sizes:", pd.Series(km_res.labels_).value_counts().sort_index().to_dict())
print("First 10 cluster labels:", km_res.labels_[:10])
print("Total within-cluster SS (inertia):", km_res.inertia_)

centers = pd.DataFrame(km_res.cluster_centers_, columns=Z.columns)
centers

# add cluster label to annotation
ann["km_cluster"] = pd.Categorical(km_res.labels_)
ann.sample(5, random_state=RANDOM_STATE)

dist_to_center = ((Z.to_numpy() - km_res.cluster_centers_[km_res.labels_]) ** 2).sum(axis=1)
withinss = pd.Series(dist_to_center).groupby(km_res.labels_).sum()
print('Within-SS for clusters:')
withinss.sort_values().round()
```



```{python}
#| label: silhouette_scores

km_res = res_kmeans[5]
sil_values = silhouette_samples(Z, km_res.labels_)
sil_mean = silhouette_score(Z, km_res.labels_)

sil_df = pd.DataFrame({
    "cluster": km_res.labels_.astype(str),
    "silhouette": sil_values,
})

plt.figure(figsize=(7, 5))
ax = sns.violinplot(
    data=sil_df,
    x="cluster",
    y="silhouette",
    hue = "cluster",
    palette="muted",
    inner=None,
    order=sorted(sil_df["cluster"].unique())
)

# Optional jittered points
sns.stripplot(
    data=sil_df,
    x="cluster",
    y="silhouette",
    color="black",
    alpha=0.3,
    jitter=True,
    size=2,
    ax=ax
)

# Overall mean
plt.axhline(sil_mean, linestyle="--", color="gray", label=f"Mean = {sil_mean:.3f}")

plt.xlabel("Cluster")
plt.ylabel("Silhouette score")
plt.title("Silhouette scores per cluster (k-Means)")
plt.legend()
plt.tight_layout()
plt.show()
```
