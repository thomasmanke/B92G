# This calculates the full cross-tabulation
X_oh = X_oh.astype(int) 

Z = (X_oh - X_oh.mean()) / X_oh.std()

CT = X_oh.T @ X_oh # = intersections = observed

counts = X_oh.sum(axis=0)

union = (
    counts[:,None]
    + counts[None,:]
    - CT
)

expected = np.outer(counts, counts) / len(X_oh)
pearson = (CT - expected) / np.sqrt(expected)
jaccard = CT / union

A = jaccard


#A = Z.T @ Z / (len(Z) - 1)

CT_df = pd.DataFrame(
    A,
    columns=encoder.get_feature_names_out(cat_cols),
    index=encoder.get_feature_names_out(cat_cols)
